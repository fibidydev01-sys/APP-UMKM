              <div className="flex items-center gap-4">
                {logo && (
                  <div className="relative h-24 w-24 md:h-28 md:w-28 rounded-3xl overflow-hidden border-3 border-zinc-400/40 shadow-2xl shadow-zinc-500/60">
                    <OptimizedImage
                      src={logo}
                      alt={storeName || title}
                      fill
                      className="object-cover grayscale"
                    />
                  </div>
                )}

                <Badge variant="secondary" className="px-5 py-2.5 text-lg bg-zinc-500/20 text-zinc-200 border-zinc-400/40">
                  <Paintbrush className="h-6 w-6 mr-2" />
                  {storeName || 'METALLIC'}
                </Badge>
              </div>

              <h1 className="text-5xl sm:text-6xl md:text-7xl lg:text-8xl font-bold text-transparent bg-clip-text bg-gradient-to-r from-zinc-300 via-zinc-100 to-zinc-300 leading-tight"
                style={{
                  textShadow: '0 0 40px rgba(255, 255, 255, 0.3)'
                }}
              >
                {title}
              </h1>

              {subtitle && (
                <p className="text-lg sm:text-xl md:text-2xl text-zinc-200/80 leading-relaxed">
                  {subtitle}
                </p>
              )}

              {showCta && ctaText && (
                <div className="pt-6">
                  <Link href={ctaLink}>
                    <InteractiveHoverButton className="min-w-[240px] text-lg px-10 py-6 bg-zinc-700 hover:bg-zinc-600 border border-zinc-500">
                      {ctaText}
                    </InteractiveHoverButton>
                  </Link>
                </div>
              )}

              {/* Metallic Stats */}
              <div className="grid grid-cols-3 gap-4 pt-8 border-t border-zinc-400/20">
                {[
                  { label: 'Texture', value: 'NOISE' },
                  { label: 'Finish', value: 'METAL' },
                  { label: 'Style', value: 'ART' }
                ].map((stat, i) => (
                  <div key={i} className="text-center">
                    <div className="text-sm text-zinc-300/60 mb-1">{stat.label}</div>
                    <div className="text-base text-zinc-200 font-bold">{stat.value}</div>
                  </div>
                ))}
              </div>
            </motion.div>

            {/* Right - Metallic Gallery */}
            <motion.div
              initial={{ opacity: 0, x: 50, scale: 0.95 }}
              animate={{ opacity: 1, x: 0, scale: 1 }}
              transition={{ duration: 0.8, delay: 0.3 }}
              className="grid grid-cols-2 gap-6"
            >
              {[1, 2, 3, 4].map((i) => (
                <motion.div
                  key={i}
                  whileHover={{ scale: 1.05, rotate: 2 }}
                  className="relative aspect-square rounded-2xl overflow-hidden border-2 border-zinc-400/30 shadow-2xl"
                >
                  {backgroundImage ? (
                    <>
                      <OptimizedImage
                        src={backgroundImage}
                        alt={`Metallic ${i}`}
                        fill
                        className="object-cover grayscale"
                      />
                      <div className="absolute inset-0 bg-gradient-to-br from-white/10 to-transparent" />
                    </>
                  ) : (
                    <div className="absolute inset-0 bg-gradient-to-br from-zinc-700 to-zinc-900 flex items-center justify-center">
                      <span className="text-6xl opacity-20 font-bold">{i}</span>
                    </div>
                  )}
                </motion.div>
              ))}
            </motion.div>
          </div>
        </div>
      </div>
    </section>
  );
}
