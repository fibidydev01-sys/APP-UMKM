                speed={1}
                enableShadows
                enableOnHover={false}
                className="text-6xl sm:text-7xl md:text-8xl lg:text-9xl xl:text-[12rem] font-black text-orange-400"
              >
                {title}
              </GlitchText>
            </motion.div>

            {/* Plasma Subtitle */}
            {subtitle && (
              <motion.p
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}

                className="text-xl sm:text-2xl md:text-3xl text-orange-100/90 text-center max-w-4xl leading-relaxed"

                  bounce: 0.4
                }}
                className="pt-8"
              >
                <Link href={ctaLink}>
                  <InteractiveHoverButton className="min-w-[300px] text-xl px-14 py-8 bg-orange-600 hover:bg-orange-500 shadow-[0_0_40px_rgba(251,146,60,0.6)]">
                    {ctaText}
                  </InteractiveHoverButton>
                </Link>
              </motion.div>
            )}

            {/* Plasma Energy Grid */}
            {backgroundImage && (
              <motion.div
                initial={{ opacity: 0, y: 50 }}

              >
                {[1, 2, 3].map((i) => (
                  <motion.div
                    key={i}
                    whileHover={{ 
                      scale: 1.1, 
                      rotate: 360,
                      boxShadow: '0 0 40px rgba(251, 146, 60, 0.8)'
                    }}
                    transition={{ type: "spring", bounce: 0.5 }}
                    className="relative aspect-square rounded-full overflow-hidden border-4 border-orange-400/50 shadow-2xl shadow-orange-500/50"
                  >
                    <OptimizedImage
                      src={backgroundImage}
                      alt={`Plasma ${i}`}
                      fill
                      className="object-cover"
                    />
                    <div className="absolute inset-0 bg-orange-600/20" />
                  </motion.div>
                ))}
              </motion.div>
            )}

            {/* Energy Indicators */}
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              transition={{ delay: 1, duration: 1 }}
              className="flex gap-12 text-orange-400/60"
            >
              {['‚ö°', 'üí•', '‚ö°', 'üí•', '‚ö°'].map((icon, i) => (
                <motion.span
                  key={i}
                  animate={{ 
                    scale: [1, 1.3, 1],
                    opacity: [0.4, 1, 0.4]
                  }}
                  transition={{
                    duration: 2,
                    delay: i * 0.2,
                    repeat: Infinity,
                    ease: "easeInOut"
                  }}
                  className="text-4xl"
                >
                  {icon}
                </motion.span>
              ))}
            </motion.div>
          </div>
        </div>
      </div>
    </section>
  );
}
