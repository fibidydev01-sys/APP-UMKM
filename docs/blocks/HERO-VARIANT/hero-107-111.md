                className="text-6xl sm:text-7xl md:text-8xl lg:text-9xl xl:text-[12rem] font-black text-gray-300 leading-none"
              >
                {title}
              </ScrollReveal>
            </motion.div>

            {/* Veil Subtitle */}
            {subtitle && (
              <motion.p
                initial={{ opacity: 0 }}

            {showCta && ctaText && (
              <motion.div

                transition={{ delay: 0.6, duration: 0.8 }}
                className="pt-8"
              >
                <Link href={ctaLink}>
                  <InteractiveHoverButton className="min-w-[280px] text-xl px-12 py-7 bg-gray-700 hover:bg-gray-600 border border-gray-500">
                    {ctaText}
                  </InteractiveHoverButton>
                </Link>
              </motion.div>
            )}

            {/* Veiled Image Gallery */}
            {backgroundImage && (
              <motion.div
                initial={{ opacity: 0, y: 50 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.8, duration: 0.8 }}
                className="grid grid-cols-2 md:grid-cols-4 gap-6 max-w-5xl mx-auto pt-12"
              >
                {[1, 2, 3, 4].map((i) => (
                  <motion.div
                    key={i}
                    whileHover={{ scale: 1.05, filter: 'grayscale(0%)' }}
                    className="relative aspect-square rounded-2xl overflow-hidden border-2 border-gray-400/30 shadow-xl grayscale hover:grayscale-0 transition-all duration-500"
                  >
                    <OptimizedImage
                      src={backgroundImage}
                      alt={`Veil ${i}`}
                      fill
                      className="object-cover"
                    />
                    <div className="absolute inset-0 bg-gray-900/20" />
                  </motion.div>
                ))}
              </motion.div>
            )}

            {/* Mystery Symbols */}
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              transition={{ delay: 1, duration: 1 }}
              className="flex gap-8 text-gray-400/40"
            >
              {['‚óÜ', '‚óá', '‚óÜ', '‚óá', '‚óÜ'].map((symbol, i) => (
                <motion.span
                  key={i}
                  animate={{ 
                    opacity: [0.2, 0.6, 0.2]
                  }}
                  transition={{
                    duration: 3,
                    delay: i * 0.3,
                    repeat: Infinity,
                    ease: "easeInOut"
                  }}
                  className="text-3xl"
                >
                  {symbol}
                </motion.span>
              ))}
            </motion.div>
          </div>
        </div>
      </div>
    </section>
  );
}
