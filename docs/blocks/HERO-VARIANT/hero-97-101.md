                    </div>
                  </motion.div>
                )}

                <Badge variant="secondary" className="px-6 py-3 text-xl bg-blue-500/20 text-blue-100 border-blue-300/40 backdrop-blur-xl">
                  <Snowflake className="h-7 w-7 mr-2" />
                  {storeName || 'WINTER'}
                </Badge>

                {/* Frosty Title */}
                <h1 className="text-6xl sm:text-7xl md:text-8xl lg:text-9xl xl:text-[12rem] font-black text-white leading-none"
                  style={{
                    textShadow: '0 0 40px rgba(147, 197, 253, 0.8), 0 0 80px rgba(147, 197, 253, 0.4), 4px 4px 0px rgba(30, 58, 138, 0.3)'
                  }}
                >
                  {title}
                </h1>

                {/* Icy Subtitle */}
                {subtitle && (
                  <p className="text-xl sm:text-2xl md:text-3xl text-blue-100/90 max-w-3xl mx-auto leading-relaxed">
                    {subtitle}
                  </p>
                )}

                {/* Frozen CTA */}
                {showCta && ctaText && (
                  <div className="pt-8">
                    <Link href={ctaLink}>
                      <InteractiveHoverButton className="min-w-[280px] text-xl px-12 py-7 bg-blue-600 hover:bg-blue-500 backdrop-blur-xl border border-blue-300/30 shadow-[0_0_30px_rgba(59,130,246,0.5)]">
                        {ctaText}
                      </InteractiveHoverButton>
                    </Link>
                  </div>
                )}

                {/* Frozen Image Gallery */}
                {backgroundImage && (
                  <motion.div
                    initial={{ opacity: 0, y: 50 }}
                    animate={{ opacity: 1, y: 0 }}
                    transition={{ delay: 0.8, duration: 0.8 }}
                    className="grid grid-cols-3 gap-4 max-w-3xl mx-auto pt-12"
                  >
                    {[1, 2, 3].map((i) => (
                      <motion.div
                        key={i}
                        whileHover={{ scale: 1.05, y: -10 }}
                        className="relative aspect-square rounded-2xl overflow-hidden border-2 border-blue-300/30 shadow-xl shadow-blue-500/30 backdrop-blur-xl bg-blue-500/10"
                      >
                        <OptimizedImage
                          src={backgroundImage}
                          alt={`Snow ${i}`}
                          fill
                          className="object-cover"
                        />
                        <div className="absolute inset-0 bg-blue-600/10" />
                      </motion.div>
                    ))}
                  </motion.div>
                )}
              </motion.div>
            </div>

            {/* Bottom Snowflakes */}
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              transition={{ delay: 1, duration: 1 }}
              className="flex justify-center gap-8 text-4xl text-blue-200/30"
            >
              {['‚ùÑ', '‚ùÖ', '‚ùÜ', '‚ùÑ', '‚ùÖ'].map((flake, i) => (
                <motion.span
                  key={i}
                  animate={{ 
                    y: [0, -20, 0],
                    rotate: [0, 360]
                  }}
                  transition={{
                    duration: 3,
                    delay: i * 0.2,
                    repeat: Infinity,
                    ease: "easeInOut"
                  }}
                >
                  {flake}
                </motion.span>
              ))}
            </motion.div>
          </div>
        </div>
      </div>
    </section>
  );
}
