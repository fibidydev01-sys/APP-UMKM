              className="lg:col-span-2 space-y-8 md:space-y-10"
            >
              <div className="flex items-center gap-4">
                {logo && (
                  <div className="relative h-24 w-24 md:h-28 md:w-28 rounded-none overflow-hidden border-4 border-purple-400/40 shadow-2xl" style={{ imageRendering: 'pixelated' }}>
                    <OptimizedImage
                      src={logo}
                      alt={storeName || title}
                      fill
                      className="object-cover"
                      style={{ imageRendering: 'pixelated' }}
                    />
                  </div>
                )}

                <Badge variant="secondary" className="px-5 py-2.5 text-lg bg-purple-500/20 text-purple-200 border-purple-400/40 font-mono">
                  <Grid2x2 className="h-6 w-6 mr-2" />
                  {storeName || 'PIXEL'}
                </Badge>
              </div>

              <h1 className="text-5xl sm:text-6xl md:text-7xl lg:text-8xl font-bold text-purple-400 leading-tight font-mono"
                style={{
                  textShadow: '4px 4px 0px rgba(0,0,0,0.8), 0 0 20px rgba(168, 85, 247, 0.5)'
                }}
              >
                {title}
              </h1>

              {subtitle && (
                <p className="text-lg sm:text-xl md:text-2xl text-purple-100/80 leading-relaxed font-mono">
                  {subtitle}
                </p>
              )}

              {showCta && ctaText && (
                <div className="pt-6">
                  <Link href={ctaLink}>
                    <InteractiveHoverButton className="min-w-[240px] text-lg px-10 py-6 bg-purple-600 hover:bg-purple-500 font-mono border-4 border-purple-400 rounded-none shadow-[8px_8px_0px_rgba(0,0,0,0.8)]">
                      {ctaText}
                    </InteractiveHoverButton>
                  </Link>
                </div>
              )}

              {/* Pixel Stats */}
              <div className="grid grid-cols-3 gap-4 pt-8 border-t border-purple-400/20">
                {[
                  { label: 'RESOLUTION', value: '8-BIT' },
                  { label: 'STYLE', value: 'RETRO' },
                  { label: 'MODE', value: 'PIXEL' }
                ].map((stat, i) => (
                  <div key={i} className="text-center font-mono">
                    <div className="text-xs text-purple-300/60 mb-1">{stat.label}</div>
                    <div className="text-lg text-purple-200 font-bold">{stat.value}</div>
                  </div>
                ))}
              </div>
            </motion.div>
          </div>
        </div>
      </div>
    </section>
  );
}
