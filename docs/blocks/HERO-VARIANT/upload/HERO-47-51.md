                      />
                    ) : (

                        <span className="text-4xl opacity-20">3</span>
                      </div>
                    )}
                  </div>
                </div>

                <div className="space-y-4 md:space-y-6 pt-8 md:pt-12">
                  <div className="relative h-32 md:h-40 rounded-2xl overflow-hidden shadow-xl border border-orange-400/20">
                    {backgroundImage ? (
                      <OptimizedImage
                        src={backgroundImage}
                        alt={`${storeName || title} 2`}
                        fill
                        className="object-cover hover:scale-110 transition-transform duration-500"
                      />
                    ) : (

                        <span className="text-4xl opacity-20">2</span>
                      </div>
                    )}
                  </div>

                  <div className="relative h-48 md:h-64 rounded-2xl overflow-hidden shadow-xl border border-orange-400/20">
                    {backgroundImage ? (
                      <OptimizedImage
                        src={backgroundImage}
                        alt={`${storeName || title} 4`}

                        <span className="text-6xl opacity-20">4</span>
                      </div>
                    )}
                  </div>
                </div>
              </motion.div>
            </div>
          </div>
        </div>
      </div>
    </section>
  );
}
