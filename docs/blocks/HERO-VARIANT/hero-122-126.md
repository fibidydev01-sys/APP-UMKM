```tsx
                  <OptimizedImage
                    src={logo}
                    alt={storeName || title}
                    fill
                    className="object-cover"
                  />
                </div>
              )}

              <Badge variant="secondary" className="px-6 py-3 text-xl bg-pink-500/20 text-pink-200 border-pink-400/50 backdrop-blur-xl">
                <ScanLine className="h-7 w-7 mr-2" />
                {storeName || 'SCAN'}
              </Badge>
            </motion.div>

            {/* Scan Title */}
            <motion.h1
              initial={{ opacity: 0, y: -50 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ duration: 0.8, delay: 0.2 }}
              className="text-6xl sm:text-7xl md:text-8xl lg:text-9xl xl:text-[12rem] font-black text-transparent bg-clip-text bg-gradient-to-r from-pink-400 via-purple-400 to-pink-400 leading-none"
              style={{
                textShadow: '0 0 60px rgba(236, 72, 153, 0.6)'
              }}
            >
              {title}
            </motion.h1>

            {/* Scan Subtitle */}
            {subtitle && (
              <motion.p
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                transition={{ delay: 0.4, duration: 0.8 }}
                className="text-xl sm:text-2xl md:text-3xl text-pink-100/90 max-w-4xl leading-relaxed"
              >
                {subtitle}
              </motion.p>
            )}

            {/* Scan CTA */}
            {showCta && ctaText && (
              <motion.div
                initial={{ opacity: 0, y: 30 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.6, duration: 0.8 }}
                className="pt-8"
              >
                <Link href={ctaLink}>
                  <InteractiveHoverButton className="min-w-[300px] text-xl px-14 py-8 backdrop-blur-xl bg-pink-600/80 hover:bg-pink-500/80 shadow-[0_0_40px_rgba(236,72,153,0.6)]">
                    {ctaText}
                  </InteractiveHoverButton>
                </Link>
              </motion.div>
            )}

            {/* Scanning Images */}
            {backgroundImage && (
              <motion.div
                initial={{ opacity: 0, y: 50 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.8, duration: 0.8 }}
                className="grid grid-cols-3 gap-6 max-w-4xl mx-auto pt-12"
              >
                {[1, 2, 3].map((i) => (
                  <motion.div
                    key={i}
                    whileHover={{ scale: 1.05, y: -10 }}
                    className="relative aspect-square rounded-3xl overflow-hidden border-2 border-pink-400/30 shadow-2xl backdrop-blur-xl"
                  >
                    <OptimizedImage
                      src={backgroundImage}
                      alt={`Scan ${i}`}
                      fill
                      className="object-cover"
                    />
                    <div className="absolute inset-0 bg-pink-600/10" />
                  </motion.div>
                ))}
              </motion.div>
            )}
          </div>
        </div>
      </div>
    </section>
  );
}
```
