## CTA-18: Split Text Animation with Magnet Button

**Layout**: Center alignment dengan split text animation  
**Animation**: SplitText untuk title, Magnet untuk button **Style**: Clean
dengan animated text reveal

```tsx
// Cta18.tsx
'use client';

import Link from 'next/link';
import { ArrowRight } from 'lucide-react';
import { Button } from '@/components/ui/button';
import SplitText from '@/components/ui/SplitText';
import Magnet from '@/components/ui/Magnet';

/**
 * Cta18 Props - Mapped from Data Contract (LANDING-DATA-CONTRACT.md)
 *
 * @prop title - ctaTitle: CTA heading
 * @prop subtitle - ctaSubtitle: CTA description
 * @prop buttonText - ctaButtonText: Button label
 * @prop buttonLink - ctaButtonLink: Button destination URL
 * @prop buttonVariant - ctaButtonStyle: 'default' | 'secondary' | 'outline'
 */
interface Cta18Props {
  title: string;
  subtitle?: string;
  buttonText: string;
  buttonLink: string;
  buttonVariant: 'default' | 'secondary' | 'outline';
}

/**
 * CTA Block: cta18
 * Design: Split Text Animation with Magnet Button
 */
export function Cta18({
  title,
  subtitle,
  buttonText,
  buttonLink,
  buttonVariant,
}: Cta18Props) {
  return (
    <section className="py-20 my-8">
      <div className="text-center max-w-4xl mx-auto px-4">
        {/* Animated Title */}
        <SplitText
          text={title}
          className="text-4xl md:text-5xl lg:text-6xl font-extrabold mb-6"
          delay={50}
          duration={1.25}
          ease="power3.out"
          splitType="chars"
          from={{ opacity: 0, y: 40 }}
          to={{ opacity: 1, y: 0 }}
          threshold={0.1}
          textAlign="center"
        />

        {subtitle && (
          <p className="text-lg md:text-xl text-muted-foreground mb-8 max-w-2xl mx-auto">
            {subtitle}
          </p>
        )}

        {/* Magnet Button */}
        <Magnet padding={50} magnetStrength={50}>
          <Link href={buttonLink}>
            <Button
              size="lg"
              variant={buttonVariant}
              className="gap-2 h-14 px-8 text-lg"
            >
              {buttonText}
              <ArrowRight className="h-5 w-5" />
            </Button>
          </Link>
        </Magnet>
      </div>
    </section>
  );
}
```
